Class saferbuf

Safe rbuf implements everything bin.rbuf does but adds bounds checks. Has following structure.

struct bin_safe_rbuf {
    const char * buf;
    union {
        const char     *c;
        const int8_t   *i8;
        const uint8_t  *u8;
        const int16_t  *i16;
        const uint16_t *u16;
        const int32_t  *i32;
        const uint32_t *u32;
        const int64_t  *i64;
        const uint64_t *u64;
    } p;
    size_t len;
};
